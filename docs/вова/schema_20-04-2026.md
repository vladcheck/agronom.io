Table TechnologicalCards AS TCards {
  ~id_index_template
  ~soft_delete_template
  enterprise_id bigint [not null, ref: > Ent.id]
  variety_id bigint [not null, ref: > V.id]
  planting_density int [not null]
  planting_norm_per_ha float [not null] // норма высадки (тыс. шт/га)
  planned_yield decimal [not null] // целевая урожайность
  growth_duration_days int [default: null] // срок выращивания рассады
  applied_seed_weight decimal [default: null] // 

  indexes {
    enterprise_id
    variety_id
  }

  checks {
    `planting_norm > 0` [name: "chk_planting_norm"]
    `planting_density > 0 AND planting_density < 1000` [name: "chk_planting_density"]
    `growth_duration_days > 0` [name: "chk_growth_duration_days"]
  }
}
